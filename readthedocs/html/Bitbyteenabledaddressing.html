property rtl_byte_enable { type = boolean; component = addrmap;};

addrmap block1 {

                        rtl_byte_enable=false;

                        reg reg1 {

                                regwidth = 32;

                                field F1 {

                                        hw = rw;

                                        sw = rw;

                                } F1[31:0] = 32'h0;

                         };reg1 reg1 @0x0;

};
