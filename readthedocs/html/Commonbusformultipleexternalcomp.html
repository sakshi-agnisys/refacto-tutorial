Common bus for multiple external components

Property sharedextbus


Property "sharedextbus" configures a common external bus. This property is applicable on block or regGroup/section.


Examples:


IDS-Word



IDS-Excel



SystemRDL


addrmap Block1 {

       name  = "Block1 Address Map";

       addrmap Reg_group1 {

               name  = "Reg_group1 Address Map";

               sharedextbus = true ;


               reg Reg1 { 

                       regwidth = 32;

                       field {

                               hw = rw;

                               sw = rw;

                               onread=r;

                               onwrite=w;

                       } Fld1[31:0] = 32'h0;

               };

 

               reg Reg2 { 

                       regwidth = 32;

                       field {

                               hw = rw;

                               sw = rw;

                               onread=r;

                               onwrite=w;

                       } Fld1[31:0] = 32'h01;

               };

 

               external Reg1 Reg1 @0x0;

               Reg2 Reg2 @0x4;

       };

       Reg_group1  Reg_group1 @0x0;

};

Support "sharedextbus" with “external_ack=optimized”


IDS provides the “sharedextbus” property which forces all external registers to share a common bus.


IDS-NG input:

                       

                     


IDS-Excel:

        

                     


SystemRDL


property external_ack { type = string; component = addrmap;};

     


  addrmap test {


    external_ack = "optimized";


    sharedextbus = true ;         


     reg reg0 { 


       regwidth = 32;


      field {


         hw = w;


         sw = r;

         onread=rclr;


      } fld[31:0] = 32'h0;


       };      


    reg reg1 { 


       regwidth = 32;


      field {


         hw = rw;


         sw = rw;


      } fld[31:0] = 32'h0;


    };   

     reg reg2 { 


      regwidth = 32;


      field {


        hw = rw;


        sw = rw;


      } fld[31:0] = 32'h0;


    };   

    reg0 reg0 @0x0;

 

    external reg1 reg1 @0x4;


    external reg2 reg2 @0x8;


};
