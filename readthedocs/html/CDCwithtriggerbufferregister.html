CDC with trigger buffer register


(compatible with IDS version 6.46.0.0 and above for Verilog)


CDC is supported with "buffer_trig" property or UDP (in case of SystemRDL). It is currently supported with following two flows:

  • Mux synchronization
  • Handshake synchronization


Detailed example of both the flows is given below:


Mux Synchronization:


SystemRDL


property cdc_clock {type = string; component = reg; };

property buffer_trig {type = string; component = reg; };

property endianness {type = string; component = reg; };

addrmap top_level {

       reg holding_reg {

               buffer_trig = "trigger";

               endianness = "little";

               regwidth = 64;

               cdc_clock = "clk1";

               field {

                       hw = rw;

                       sw = r;

               } trigger[31:0] = 32'h0;

               field {

                       hw = rw;

                       sw = r;

               } buffer[63:32] = 32'h0;

       };

       holding_reg holding_reg;

} ;

input clk1;        // Clock domain clk1

// HW WRITE-ABLE SIGNAL FOR EACH FIELD

    input   holdingreg_trigger_in_enb ;     

    reg   holdingreg_trigger_in_enb_ff0;    

    reg   holdingreg_trigger_in_enb_ff1;     

    reg   holdingreg_trigger_in_enb_ff2;     

    input   holdingreg_buffer_in_enb  ;      

    reg   holdingreg_buffer_in_enb_ff0;      

    reg   holdingreg_buffer_in_enb_ff1;     

    reg   holdingreg_buffer_in_enb_ff2;     

    

    // READ DATA SIGNAL FOR EACH FIELD

    output  [31 : 0] holdingreg_trigger_r ;     

    reg  [31 : 0] holdingreg_trigger_r_ff0 ;     

    reg  [31 : 0] holdingreg_trigger_r_ff1 ;     

    output  [31 : 0] holdingreg_buffer_r  ;     

    reg  [31 : 0] holdingreg_buffer_r_ff0  ;      

    reg  [31 : 0] holdingreg_buffer_r_ff1  ;      

    

    // HW WRITE DATA SIGNAL FOR EACH FIELD

    input   [31 : 0] holdingreg_trigger_in ;      

    reg   [31 : 0] holdingreg_trigger_in_ff0 ;      


    input   [31 : 0] holdingreg_buffer_in  ;      

    reg   [31 : 0] holdingreg_buffer_in_ff0  ;


// BUFFER : HOLDINGREG_BUFF_0   SIGNALS

    reg [31:0] holdingreg_buff_0 ;

    wire [64 - 1 : 0] holdingreg_buff_0_offset;

    wire holdingreg_buff_0_decode;

    wire [31:0] holdingreg_buff_0_rd_data;


// Hardware Clock Domain flop

always @(posedge clk1)

        begin

        if (!reset_l)

            begin

                holdingreg_trigger_in_enb_ff2 <= 1'b0 ;

                holdingreg_trigger_in_ff0 <= 32'b00000000000000000000000000000000;

                holdingreg_trigger_r_ff0 <= 32'b00000000000000000000000000000000;

                holdingreg_trigger_r_ff1 <= 32'b00000000000000000000000000000000;

            end

        else

            begin

                holdingreg_trigger_in_enb_ff2 <= holdingreg_trigger_in_enb;

                holdingreg_trigger_in_ff0 <= holdingreg_trigger_in_enb ? holdingreg_trigger_in : holdingreg_trigger_in_ff0;

// 2-DFF Synchronizer

                holdingreg_trigger_r_ff1 <= holdingreg_trigger_r_ff0;

                holdingreg_trigger_r_ff0 <= holdingreg_trigger_q;

            end

    end // always clk


// Mux synchronizer in trigger buffer register flop (Software Clock Domain)

                holdingreg_trigger_in_enb_ff1 <= holdingreg_trigger_in_enb_ff2 ;

                holdingreg_trigger_in_enb_ff0 <= holdingreg_trigger_in_enb_ff1 ;

            if (holdingreg_trigger_in_enb_ff0)   // TRIGGER : HW Write

                begin

                    holdingreg_trigger_q <= holdingreg_trigger_in_ff0;

                end


Handshake Synchronization:


SystemRDL


property cdc_clock {type = string; component = reg; };

property buffer_trig {type = string; component = reg; };

property endianness {type = string; component = reg; };

addrmap top_level {

       reg holding_reg {

               buffer_trig = "trigger";

               endianness = "little";

               regwidth = 64;

               cdc_clock = "clk1:handshake";

               field {

                       hw = rw;

                       sw = r;

               } trigger[31:0] = 32'h0;

               field {

                       hw = rw;

                       sw = r;

               } buffer[63:32] = 32'h0;

       };

       holding_reg holding_reg;

} ;

    input   holdingreg_buffer_in_enb  ;      

    wire   holdingreg_buffer_in_enb_ff;      

    // BUFFER SIGNAL FOR EACH FIELD

    reg [31 : 0] holdingreg_trigger_q ;      

    reg [31 : 0] holdingreg_trigger_q_ff ;

    wire   holdingreg_trigger_read_enb ;

    wire   holdingreg_trigger_read_busy_out ;

    reg [31 : 0] holdingreg_buffer_q  ;      

    reg [31 : 0] holdingreg_buffer_q_ff  ;

    wire   holdingreg_buffer_read_enb  ;

    wire   holdingreg_buffer_read_busy_out  ; 

    

    // HW WRITE DATA SIGNAL FOR EACH FIELD

    input   [31 : 0] holdingreg_trigger_in ;      

    output    holdingreg_trigger_busy_out ;

    wire   [31 : 0] holdingreg_trigger_in_ff ;      

    input   [31 : 0] holdingreg_buffer_in  ;      

    output    holdingreg_buffer_busy_out  ;

    wire   [31 : 0] holdingreg_buffer_in_ff  ;      // FIELD : BUFFER


// Software clock domain – Trigger Field

if (holdingreg_trigger_in_enb_ff)   // TRIGGER : HW Write

                begin

                    holdingreg_trigger_q <= holdingreg_trigger_in_ff;

                end


// Software clock domain – Buffer Field

if (holdingreg_buffer_in_enb_ff)   // BUFFER : HW Write

                begin

                    holdingreg_buffer_q <= holdingreg_buffer_in_ff;

                end


// Handshake synchronizer module instantiation for write process

handshake_synchronizer_write #(.field_width(32)) holdingreg_trigger_write(.clk(clk),.reset_l(reset_l),.hw_clk(clk1),

    .register_fld_in(holdingreg_trigger_in),

    .register_fld_in_ff(holdingreg_trigger_in_ff),

    .register_fld_in_enb(holdingreg_trigger_in_enb),

    .register_fld_in_enb_ff(holdingreg_trigger_in_enb_ff),
