Vectored hardware port

In IDS as default, ports for each field in a register are created. But to have a single port for driving input data to the register and a single port to drive output from a register rather than creating individual ports for each field, vectored hardware port can be used.


This can be achieved by using the property “rtl_hw_vector” and setting it to true which would enable the user to write on each field of a register using a single input and read from each using a single output collectively. The data will split into sub-parts in accordance to the field size.


Using these new ports only the HW side will be impacted and SW side will remain the same.


From 6.42.0.0 onwards, in “rtl_hw_vector” property input and registered output ports of registers will now be created as per the size of registers, regardless of the fields being reserved or used. To avoid the lint warning in synthesis user can use “no_lint_warn” parameter.




 fig 1- The input data gets split into two parts in accordance to the field Bits.





fig 2- The output data from each field gets concatenated and the given as output.

SystemRDL


property rtl_hw_vector {type = boolean; component = reg|regfile|addrmap;};

addrmap Block_1 {

 reg reg1 { 

     rtl_hw_vector = true;

    field {

      hw = rw;

      sw = rw;

      

    } F1[31:16] = 16'h0;

    field {

      hw = rw;

      sw = rw;

      

    } F2[15:0] = 16'h0;


  };

 

  reg1 reg1 @0x0;


};

    constant default_reg1 : std_logic_vector (31 downto 0) := "00000000000000000000000000000000";

    constant offset_reg1 : unsigned  (63 downto 0) := C_Block1_offset + "0";  -- offset value

    

    

    

    -- all fields writeable by HW

    type Block1_inrec is record

        --: REG1

        reg1 : std_logic_vector(31 downto 0);

    end record;

    

   .       .      .

   .       .      .

   .       .      .

   .       .      .

    reg1_F1 :  process (clk)

        begin

            if  rising_edge (clk) then

                if reset_l = '0' then

                    q_reg1_F1 <= default_reg1(31 downto 16);

                else

                    if (reg_in_enb.reg1(0) = '1' )  then        --  HW write

                        q_reg1_F1 <= reg_in.reg1(31 downto 16);

                        

                    else

                        if ((wr_valid_reg1) = '1'  ) then         --  SW Write

           q_reg1_F1 <= ( wr_data(31 downto 16)
          and reg_enb(31     downto 16) ) or (q_reg1_F1 and 
          (not(reg_enb(31 downto 16))));

                            

                        end if;

                    end if;

                end if;-- reset

            end if;-- clock edge

        end process reg1_F1;

        -- End REG1_F1 process
