Multi-Dimensional Registers

Multi Dimensional Registers

"repeat" Property 


"repeat" property on chip/block/section/registers repeats the template the number of times as specified in the property. This property enhances the template scope in one-dimension or multi-dimensions depending upon its usage.


One-Dimensional Array


Usage: repeat=<any positive integer value>

addrmap block {

  reg reg1 {

    field {

  };

Verilog Output


module Block_IDS(

    

    // REGISTER : REG PORT SIGNAL

    Reg_enb,

    Reg_fld_r,

    

    

    

    //BUS signals

   .

   .

   .

    );

    

    

    // REGISTER : REG

    parameter Reg_count = 4;

   

    // REGISTER : REG SIGNALS

    

    

    wire    Reg_decode[Reg_count-1 : 0];        // DECODE

    wire    [Reg_count-1 : 0]Reg_wr_valid;      // WRITE VALID

    wire    [bus_width-1 : 0] Reg_rd_data[Reg_count-1 : 0];          // READ DATA

    wire   [Reg_address_width-1 : 0] Reg_offset[Reg_count-1 : 0];  // OFFSET

    wor     [bus_width-1 : 0] Reg_rd_data_wor;

    output [Reg_count-1 : 0] Reg_enb;    // REGISTER ENABLE

    

    // BUFFER SIGNAL FOR EACH FIELD

    reg [31 : 0] Reg_fld_q[Reg_count-1 : 0] ;      // FIELD : FLD

    

    // READ DATA SIGNAL FOR EACH FIELD

    output  [Reg_count*32-1 : 0] Reg_fld_r ;      // FIELD : FLD

    

    //-----------------------------------------------------

    

  

    generate

        genvar Reg_i;

        for( Reg_i = 0; Reg_i < Reg_count; Reg_i = Reg_i + 1)

        begin : Reg_gen

 

            assign Reg_wr_valid[Reg_i] = Reg_decode[Reg_i] && wr_stb;

            assign Reg_enb[Reg_i]      = Reg_wr_valid[Reg_i];

            assign Reg_offset[Reg_i] = block_offset+'h0 + Reg_i * 'h4;

            assign Reg_decode[Reg_i] = (address[Reg_address_width-1 : 0]  == Reg_offset[Reg_i]) ? 1'b1 : 1'b0;

            

            always @(posedge clk)

                begin

                if (!reset_l)

                    begin

                        Reg_fld_q[Reg_i] <= 32'b00000000000000000000000011111111;

                    end

                else

                    begin

                        //----------------------------------------------------------------------------

                        // FIELD  : FLD

                        // HW ACCESS  :  READ-ONLY                           WIDTH  :  32

                        // SW ACCESS  :  READ-WRITE                          OFFSET :  0

                        //-----------------------------------------------------------------

                        // DESCRIPTION   :  NA

                        //

                        

                    if (Reg_wr_valid[Reg_i])   // FLD : SW Write

                        begin

                            Reg_fld_q[Reg_i] <= ( wr_data[31 : 0] & reg_enb[31 : 0] ) | (Reg_fld_q[Reg_i] & (~reg_enb[31 : 0]));

                        end

                        

                        //----------------------------------------------------------------------------

                        

                        

                    end

            end // always clk

            

            

            // ----------------------------------------------------------------

            // HW OUTPUT READ DATA FOR EACH FIELD

            assign Reg_fld_r[(Reg_i) *32+31 :  (Reg_i) *32 ] =  Reg_fld_q[Reg_i] ;    // Field : FLD

            

            // ----------------------------------------------------------------

            assign Reg_rd_data[Reg_i] = Reg_decode[Reg_i] ? {Reg_fld_q[Reg_i]} : 32'b00000000000000000000000000000000;

            assign Reg_rd_data_wor = Reg_rd_data[Reg_i];

            

            

        end //Reg_gen

    endgenerate

    

    .

    .

    

endmodule

Verilog Output


module Block_IDS(

    

    // REGISTER : REG PORT SIGNAL

    Reg_enb,

    Reg_fld_r,

    

    

    //AMBA signals

    .

    .

    .

    );

    

    

    // REGISTER : REG1

    parameter Reg1count         = 4;

    parameter Reg1address_width = addr_width;

    

    // REGISTER : REG2

    parameter Reg2count         = 3;

    parameter Reg2address_width = addr_width;

    

    // REGISTER : REG3

    parameter Reg3count         = 2;

    parameter Reg3address_width = addr_width;

    

    //-----------------------------------------------------

    // REGISTER : REG SIGNALS

    

    

    wire    Reg_decode[Reg1count-1 : 0][Reg2count-1 : 0][Reg3count-1 : 0];        // DECODE

    wire    [Reg1count*Reg2count*Reg3count-1 : 0]Reg_wr_valid;      // WRITE VALID

    wire    [bus_width-1 : 0] Reg_rd_data[Reg1count-1 : 0][Reg2count-1 : 0][Reg3count-1 : 0];          // READ DATA

    wire   [Reg3address_width-1 : 0] Reg_offset[Reg1count-1 : 0][Reg2count-1 : 0][Reg3count-1 : 0];  // OFFSET

    wor     [bus_width-1 : 0] Reg_rd_data_wor;

    output [Reg1count*Reg2count*Reg3count-1 : 0] Reg_enb;    // REGISTER ENABLE

    

    // BUFFER SIGNAL FOR EACH FIELD

    reg [31 : 0] Reg_fld_q[Reg1count-1 : 0][Reg2count-1 : 0][Reg3count-1 : 0] ;      // FIELD : FLD

    

    // READ DATA SIGNAL FOR EACH FIELD

    output  [Reg1count*Reg2count*Reg3count*32-1 : 0] Reg_fld_r ;      // FIELD : FLD

    

    //-----------------------------------------------------

    

    

   .

   .

   .

   .

    

    generate

        genvar Reg1i;

        for( Reg1i = 0; Reg1i < Reg1count; Reg1i = Reg1i + 1)

        begin : Reg1gen

            

            genvar Reg2i;

            for( Reg2i = 0; Reg2i < Reg2count; Reg2i = Reg2i + 1)

            begin : Reg2gen

                

                genvar Reg3i;

                for( Reg3i = 0; Reg3i < Reg3count; Reg3i = Reg3i + 1)

                begin : Reg3gen   

                    

                    assign Reg_wr_valid[Reg1i * Reg2count * Reg3count + Reg2i * Reg3count +  Reg3i] = Reg_decode[Reg1i][Reg2i][Reg3i] && wr_stb;

                    assign Reg_enb[Reg1i * Reg2count * Reg3count + Reg2i * Reg3count +  Reg3i]      = Reg_wr_valid[Reg1i * Reg2count * Reg3count + Reg2i * Reg3count +  Reg3i];

                    assign Reg_offset[Reg1i][Reg2i][Reg3i] = block_offset+ + 'h0 + Reg1i * 'h18 + 'h0 + Reg2i * 'h8 + 'h0 + Reg3i * 'h4 + 'h0;

                    assign Reg_decode[Reg1i][Reg2i][Reg3i] = (address[Reg3address_width-1 : 0]  == Reg_offset[Reg1i][Reg2i][Reg3i]) ? 1'b1 : 1'b0;

                    

                    always @(posedge clk)

                        begin

                        if (!reset_l)

                            begin

                                Reg_fld_q[Reg1i][Reg2i][Reg3i] <= 32'b00000000000000000000000011111111;

                            end

                        else

                            begin

                               if (Reg_wr_valid[Reg1i * Reg2count * Reg3count + Reg2i * Reg3count +  Reg3i])   // FLD : SW Write

                                begin

                                    Reg_fld_q[Reg1i][Reg2i][Reg3i] <= ( wr_data[31 : 0] & reg_enb[31 : 0] ) | (Reg_fld_q[Reg1i][Reg2i][Reg3i] & (~reg_enb[31 : 0]));

                                end

                                

                                //----------------------------------------------------------------------------

                                 

                            end

                    end // always clk

                    

                    

                    // ----------------------------------------------------------------

                    // HW OUTPUT READ DATA FOR EACH FIELD

                    assign Reg_fld_r[(Reg1i * Reg2count * Reg3count + Reg2i * Reg3count +  Reg3i) *32+31 :  (Reg1i * Reg2count * Reg3count + Reg2i * Reg3count +  Reg3i) *32 ] =  Reg_fld_q[Reg1i][Reg2i][Reg3i] ;    // Field : FLD

                    

                    // ----------------------------------------------------------------

                    assign Reg_rd_data[Reg1i][Reg2i][Reg3i] = Reg_decode[Reg1i][Reg2i][Reg3i] ? {Reg_fld_q[Reg1i][Reg2i][Reg3i]} : 32'b00000000000000000000000000000000;

                    assign Reg_rd_data_wor = Reg_rd_data[Reg1i][Reg2i][Reg3i];

                    

                    

                end //Reg3gen

                

            end //Reg2gen

            

        end //Reg1gen

    endgenerate

    

    .

    .

    .

    

endmodule



Packed Array in RTL:


With the help of -rtl_array "pack" switch in command line, we can make the signals (input, output or wire) in the packed array form. It is supported through command line from the top and is supported for RTL output. It gives us the multi-dimensional packed array in RTL output ans it also support for the top_property named rtl_pack_array=true for packed array. Fixed packed arrays in case of hw=w and sw=r.


SystemRDL Input :


property chip{type=boolean; component=addrmap;};

addrmap block{

regfile sec1{

regfile sec2{

reg regB{

regwidth=32;

field{sw=rw;hw=rw;fieldwidth=32;}f1;

};

regB regB[4];

};

sec2 s2[5];

};

sec1 s1[2];

};

addrmap chip1{

chip=true;

block block1[2];

};

idsbatch test.rdl -out verilog -bus apb -dir out -rtl_array “pack” -if -preserve
