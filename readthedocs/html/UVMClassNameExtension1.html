UVM Class Name Extension

1. uvm_class


User need to mention the property name ‘uvm_class’ in the Template of the element followed by the name of the class ‘user_class_name’ e.g. ‘my_class_reg’. 


Example for user-defined class is:


Example: IDS-NG        IDS-Word        IDS-Excel        SystemRDL


IDS-NG Register View



IDS-NG Spreadsheet View



SystemRDL


property uvm_class { type = string; component = addrmap | regfile | reg; };

 

addrmap block1 {



  reg reg1 { 

     uvm_class = "my_class_reg";

     regwidth = 8;

    field {

      hw = r;

      sw = rw;

    

    } fld_1[7:0] = 8'h0;


  };

 

  reg1 reg1 @0x0;


};


Generated UVM Ouptput


class Block1_Reg1 extends my_class_reg;


rand uvm_reg_field Fld;


function new(string name = "Block1_Reg1");

 super.new(name, 32, build_coverage(UVM_NO_COVERAGE));

 add_coverage(build_coverage(UVM_NO_COVERAGE));            //Added because build coverage in the line above doesn’t work due to a bug in UVM 1.1 library

endfunction

    

virtual function void build();

  this.Fld = uvm_reg_field::type_id::create("Fld");

  this.Fld.configure(this, 32,  0, "RW", 0, 'd0, 1, 1, 0);

endfunction

    

`uvm_object_utils(Block1_Reg1)

endclass


2. uvm_base_class


This property customizes the class which is extended by the generated class.


Example: IDS-NG        IDS-Word        IDS-Excel        SystemRDL


IDS-NG Register View



IDS-NG Spreadsheet View



SystemRDL


property uvm_base_class { type = string; component = addrmap | regfile | reg; };

addrmap chip1 {

  name = "chip1 Address Map";

    uvm_base_class = "foo_chip";


  addrmap block1 {

    name  = "block1 Address Map";

    uvm_base_class = "foo_block";


    reg reg1 { 

       uvm_base_class = "foo_reg";

       regwidth = 32;

      field {

        hw = rw;

        sw = rw;

        

      } f1[31:0] = 32'h0;


    };

 

    reg reg2 { 

       regwidth = 32;

      field {

        hw = rw;

        sw = rw;

        

      } f1[31:0] = 32'h0;


    };

 

    reg1 reg1 @0x0;


    reg2 reg2 @0x4;


  };

  block1  block1 @0x0; 

    };

3. uvm.inst_class


User need to mention the property name ‘uvm.inst_class’ in the Template of the element followed by the name of the inst class ‘user_inst.class_name’ e.g. ‘inst_class_reg’. 


Example: IDS-NG        IDS-Word        IDS-Excel        SystemRDL


IDS-NG Register View



IDS-NG Spreadsheet View



SystemRDL


property uvm_inst_class { type = string; component = addrmap | regfile | reg | field; };


addrmap block1 {

 

  reg reg1 { 

     regwidth = 32;

    field {

      uvm_inst_class = "FOO";

      hw = rw;

      sw = rw;


    } f1[31:0] = 32'h0;


  };

 

  reg1 reg1 @0x0;


};


Generated UVM Output


class Block1_block extends uvm_reg_block;

  rand inst_class_reg Reg1;

    

 function new(string name = "Block1_block");

    super.new(name, UVM_NO_COVERAGE);

 endfunction


    

virtual function void build();

//create

   Reg1   =   inst_class_reg::type_id::create("Reg1");

  

//config

   Reg1.configure(this, null, "Reg1");

  

//build

   Reg1.build();

  

//define default map and add reg/regfiles

   default_map= create_map("default_map", 'h0, 4, UVM_BIG_ENDIAN, 0);

   default_map.add_reg(Reg1, 'h0, "RW");

    

   lock_model();

  

endfunction


`uvm_object_utils(Block1_block)

    

endclass : Block1_block


4. uvm.block_class:


With the “uvm.block_class” property, the chip and block class can be extended by the class mentioned as the argument to the property.


Example: IDS-NG        IDS-Word        IDS-Excel        SystemRDL


IDS-NG Register View



IDS-NG Spreadsheet View

SystemRDL

// chip : chip1


addrmap chip1 {


    uvm_block_class = "string" ;


  addrmap block1 {

   


    addrmap sec1 {

     


      reg reg1 { 

         regwidth = 32;

          hw = rw;

          sw = rw;

         :

      };

    };

8. uvm_class_name_inst


Property “uvm_class_name_inst” is created to maintain the backward compatibility of 7x series with 6x series. In 7x series (before 7.8.0.0), the uvm classes were generated with the referred block name, when “uvm_class_name_inst” isn’t applied, but when generated with 6x series, all the classes are generated with the instance block name. To match this difference in the naming of UVM classes, uvm_class_name_inst property can be applied at the top. This property is hierarchical by nature.


Example


In the example shown below, a block is referred two times under the chip. And output with and without uvm_class_name_inst property is shown.


Example: IDS_NG        IDS-Word        IDS-Excel        SystemRDL


IDS-NG Register View



IDS-NG Spreadsheet View



SystemRDL


property refpath {type=string;component=addrmap;};

property uvm_class_name_inst {type=string;component=addrmap;};

addrmap chip{

   uvm_class_name_inst="true";

   addrmap block1{

   refpath="ref.rdl:block";

   };


  addrmap block2{

   refpath="ref.rdl:block";

   };

 block1 block1_inst;

 block2 block2_inst;

};


Generated UVM output:

  `define CLASS_block1inst_block

      // Function : new

          super.new(name, UVM_NO_COVERAGE);

      endfunction

          //define default map and add reg/regfiles
