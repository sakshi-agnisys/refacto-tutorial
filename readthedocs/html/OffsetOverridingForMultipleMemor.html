Offset Overriding For Multiple Memory Maps

“uvm_add_regmap_offset” Property


Introduction


This property is used with the “uvm_add_regmap” property, using this we can change the offset of the regmap generated by “uvm_add_regmap”. The syntax to use this property is uvm_add_regmap_offset=”<name>:<base address>”.


SystemRDL Input


property uvm_add_regmap {type=string; component=addrmap;};

property uvm_add_regmap_offset {type=string; component=addrmap;};  

property chip {type=boolean; component=addrmap;};

reg regA {

  field {}F1;

};

addrmap block_1 {

    regA regA;

};

addrmap block_2 {

    regA regA;

};

addrmap block_3 {

    regA regA;

};

addrmap chip {

  chip=true;

  uvm_add_regmap     = "map1:0x40000";

  block_1 b1[2] @0x10;

  b1->uvm_add_regmap_offset = "map1:0x20";

  block_2 b2 @0x30;

  block_3 b3 @0x40;

};

addrmap top {

  chip=true;

  uvm_add_regmap     = "map1:0x40000,map2:0x200000";

    chip c1 @30000;

  c1->uvm_add_regmap_offset = "map2:0x200,map1:0x300";

  };
