Interrupts Per Channel


(Compatible with IDS version 5.20.14.0 and above)


IDesignSpec™ previously supported a single bit "irq" port which is used for representing interrupts from a block. The “irq” port is ORing of all the interrupt channels inside a block as shown below.



Sometimes it is required to have multiple interrupt outputs from the block as shown below-



This feature is supported by IDS using a property “intr.irq_per_channel”. Refer the below examples for more understanding-


Example1



Verilog Code


module block_IDS(

   ...

   ...    

    channel1_irq,     // Output interrupt signal

    channel2_irq,     // Output interrupt signal

    

    

    //Bus signals

    );

    

    output channel1_irq;     // Output interrupt signal

    

    output channel2_irq;     // Output interrupt signal

    ...    

    ...

    

    

    //Bus signals

    ...

    ...

    

    always @(posedge clk)

        

        begin

        if (!reset_l)

            begin

                Status_reg1_F1_q  <= 16'b0000000000000000;

            end

        else

            begin

                

            if (Status_reg1_wr_valid)  // acknowledging  status

                begin

                    Status_reg1_F1_q <= Status_reg1_F1_q & ( ~( wr_data[15 : 0] & reg_enb[15 : 0] ));

                end

            else

                begin

                if (Status_reg1_F1_in_enb)   // hw driven interrupts

                    begin

                        Status_reg1_F1_q <=  Status_reg1_F1_in | Status_reg1_F1_q;

                    end

                end

                

            end

    end // always clk

    

   ....

   ....

    

    

   

    

    //----------------------------------------------------------------------------

    // FIELD  : F1

    // HW ACCESS  :  READ-WRITE                          WIDTH  :  16

    // SW ACCESS  :  READ-WRITE ONE TO CLEAR             OFFSET :  0

    //-----------------------------------------------------------------

    // DESCRIPTION   :  // channel1 interrupt

    //

    

    always @(posedge clk)

        

        begin

        if (!reset_l)

            begin

                Status_reg2_F1_q  <= 16'b0000000000000000;

            end

        else

            begin

                

            if (Status_reg2_wr_valid)  // acknowledging  status

                begin

                    Status_reg2_F1_q <= Status_reg2_F1_q & ( ~( wr_data[15 : 0] & reg_enb[15 : 0] ));

                end

            else

                begin

                if (Status_reg2_F1_in_enb)   // hw driven interrupts

                    begin

                        Status_reg2_F1_q <=  Status_reg2_F1_in | Status_reg2_F1_q;

                    end

                end

                

            end

    end // always clk


    ...

    ...    

    

    assign channel1_irq  = ( |(Status_reg1_F1_q & enable_reg1_Fld1_q)) ;

    

    assign channel2_irq  = ( |(Status_reg2_F1_q & enable_reg2_Fld1_q)) ;

    

endmodule


Example 2


There are some scenarios where user need to club more than one interrupts. This functionality can be supported using "Signals" in IDS. As shown below.





Verilog Code


module Block1_IDS(

    ...

       ...

    //Signals  : SIGNALS

    block_irq1,

    block_irq2,

    block_irq,

    ...

       ...    

    //Bus signals

       ...

   );

    //  PARAMETERS

    ...    

    //Signals  : SIGNALS

    

    // Desc: BLOCK_IRQ1

    //     To assign an interrupt to this output signal, the channel name is mandatory followed by @intr.

    

    output block_irq1;

    

    // Desc: BLOCK_IRQ2

    //     To create a flopped output, the "next" property can be used.

    

    output reg block_irq2;

    

    // Desc: BLOCK_IRQ

    //

    //                  To club more channels user can use Logical OR "|" operator.

    

    output block_irq;

    

    ...

       ...

    

    always @(posedge clk)

        

        begin

        if (!reset_l)

            begin

                Status_reg1_F1_q  <= 16'b0000000000000000;

            end

        else

            begin

                

            if (Status_reg1_wr_valid)  // acknowledging  status

                begin

                    Status_reg1_F1_q <= Status_reg1_F1_q & ( ~( wr_data[15 : 0] & reg_enb[15 : 0] ));

                end

            else

                begin

                if (Status_reg1_F1_in_enb)   // hw driven interrupts

                    begin

                        Status_reg1_F1_q <=  Status_reg1_F1_in | Status_reg1_F1_q;

                    end

                end

                

            end

    end // always clk

    

   ...

   ...

    

    always @(posedge clk)

        

        begin

        if (!reset_l)

            begin

                Status_reg2_F1_q  <= 16'b0000000000000000;

            end

        else

            begin

                

            if (Status_reg2_wr_valid)  // acknowledging  status

                begin

                    Status_reg2_F1_q <= Status_reg2_F1_q & ( ~( wr_data[15 : 0] & reg_enb[15 : 0] ));

                end

            else

                begin

                if (Status_reg2_F1_in_enb)   // hw driven interrupts

                    begin

                        Status_reg2_F1_q <=  Status_reg2_F1_in | Status_reg2_F1_q;

                    end

                end

                

            end

    end // always clk

    

    ....

    ....

    assign block_irq1 = ( |(Status_reg1_F1_q & enable_reg1_Fld1_q))  ;

    

    

    

    always @(posedge clk)

        begin

        if (!reset_l)

            begin

                block_irq2 <= 1'b0;

            end

        else

            begin

                block_irq2 <= ( |(Status_reg2_F1_q & enable_reg2_Fld1_q)) ;

            end

        end // end always

        

        

        assign block_irq = ( |(Status_reg1_F1_q & enable_reg1_Fld1_q))  | ( |(Status_reg2_F1_q & enable_reg2_Fld1_q))  ;

        

    endmodule
