Enable Signal Generation

Enable for Register


In IDS a <reg_name>_enb output port is used to determine on which register the data is written from SW side, i.e., when the field is HW writeable and is designated as "<reg>_<field>_in_enb," it creates an input enable signal on the hardware interface if true. When applied at the top, if false, prevents the creation of the "<reg>_<field>_in_enb" signal for the entire regmap.


Property:

rtl.reg_enb= <true/false>


Example:        IDS-NG        IDS-Word        IDS-Excel        SystemRDL

IDS-NG Spreadsheet View

property rtl_reg_enb {type =boolean; component = addrmap|reg ; };

 

addrmap block_top {

  name  = "block_top Address Map";


  reg reg_1 { 

     rtl_reg_enb = false ;

     regwidth = 32;

    field {

      hw = rw;

      sw = rw;

      

    } R1[31:0] = 32'h0;


  };

};

Enable for Field


For getting the similar behavior for field, i.e., an output enable signal for determining on which field the data is being written from SW side, we can use the property  “rtl.field_enb”. This property works hierarchically with value as true/false. By default, it is false. This property will work with or without the multi-out option.


Property:

rtl.field_enb= <true/false> 


Example:        IDS-NG        IDS-Word        IDS-Excel        SystemRDL


IDS-NG Register View



IDS-NG Spreadsheet View


SystemRDL


property rtl_field_enb {type =boolean; component = addrmap|regfile|reg|field ; };

property registered {type =string; component = reg | field ; };

               

addrmap block_1 {

  name  = "block_1 Address Map";


  reg Reg1 { 

     regwidth = 32;

    field {

      rtl_field_enb = true ;

      hw = rw;

      sw = rw;

      


    } F1[31:16] = 16'h0;

    field {

      rtl_field_enb = true ;

      hw = rw;

      sw = rw;

      


    } F2[15:0] = 16'h0;


  };

 

  reg Reg2 { 

     rtl_field_enb = true ;

     registered = "false" ;

     regwidth = 32;

    field {

      registered = "false" ;

      hw = rw;

      sw = rw;

      

    } F1[31:16] = 16'h0;

    field {

      registered = "false" ;

      hw = rw;

      sw = rw;

      

    } F2[15:0] = 16'h0;


  };

 

  Reg1 Reg1 @0x0;


  Reg2 Reg2 @0x4;


};

    if(!reset_l)

        begin

            Reg1_parity_reg <= ~parity_gen({32'd0});

        end

    else

        begin

        if(Reg1_wr_valid)

            begin

                Reg1_parity_reg <= ~parity_gen({(wr_data[31:0] & reg_enb[31:0])});

property rtl_hw_enb {type =boolean; component = addrmap|reg|field;};

property cdc_reset {type = string; component = addrmap;};

              

 

addrmap block1_csr {

  cdc_reset = "hw_rst" ;


  reg normal { 

    field {

      hw = na;

      sw = rw;

      

    } F1[31:0] = 32'h0;


  };

 

  reg data { 

    field {

      cdc_clock = "hw_clock" ;

      hw = r;

      sw = rw;

      


    } F1[31:0] = 32'h0;


  };

 

  reg status { 

    field {

      cdc_clock = "hw_clock:handshake" ;

      rtl_hw_enb = false ;

      rtl_precedence = "sw" ;

      hw = w;

      sw = rw;

      woclr = true;


    } F1[31:0] = 32'h0;


  };

 

  normal normal @0x0;


  data data @0x4;


  status status @0x8;


};
