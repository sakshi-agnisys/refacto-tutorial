FIFO

(Compatible with IDS version 6.0 and above)


IDS provides the FIFO (first in first out) functions through the parameterized single clock IP cores. In which, simultaneous read and write operations are possible. The access supported for FIFO are ro (read-only) and wo(write-only).

Note: Fifo is supported for amba-ahb, axi, apb , wishbone and amba3ahb-lite bus only. The depth of fifo should be in 2 ^ power. An extra hardware port is created which is used to show the spaces occupied in the fifo. 


The flags of fifo are: 

  • empty – this flag shows that fifo is empty and nothing can be read.
  • full – it shows that fifo is full and nothing can be write.
  • almost_empty – shows that left spaces in fifo is equal to or less than the almost empty value given by user (this flag is only created when you specify the almost empty value by the use of property fifo.almost_empty_val).
  • almost_full - shows that occupied spaces in fifo is equal to or greater than the almost full value given by user (this flag is only created when you specify the almost full value by the use of property fifo.almost_full_val).

                  

The properties that can be used in fifo are:

  1. fifo.almost_full_val :- used to give the value the threshold value

Its value can be

  • Any register field
  • Any input signal
  • Direct value

Please note that if you don’t use this property, almost full flag will not be created.

  1. fifo.almost_full :- this property is used to send the almost full flag to any one of the following

Its value can be

  • Any register field
  • Any output signal

By default bus error (if sw = wo) or hardware port( if hw=wo)

  1. fifo.almost_empty_val :- used to give the value the threshold value

Its value can be

  • Any register field
  • Any input signal
  • Direct value

Please note that if you don’t use this property, almost full empty will not be created.

  1. fifo.almost_empty :-  this property is used to send the almost empty to any one of the following

Its value can be

  • Any register field
  • Any output signal

By default bus error (if sw = ro) or hardware port( if hw=ro)

  1. fifo.empty :- this property is used to send the empty to any one of the following

Its value can be

  • Any register field
  • Any output signal

By default bus error (if sw = ro) or hardware port( if hw=ro)

  1. fifo.full :- this property is used to send the full flag to any one of the following

Its value can be

  • Any register field
  • Any output signal

By default bus error (if sw = wo) or hardware port( if hw=wo).

module Block1_IDS(

    // FIFO REGISTER : FIFO_NAME PORT SIGNAL

    fifo_name_in,

    fifo_name_in_enb,

    fifo_name_usedW,

    fifo_name_full

    );

    //  PARAMETERS

    parameter fifo_name_depth = 8;

    …

    reg     [3:0]fifo_name_usedW_sig;

    reg     [31:0]fifo_fifo_name[fifo_name_depth-1:0];

    reg     [2:0]fifo_name_wr_ptr;        //WRITE POINTER

    reg     [2:0]fifo_name_rd_ptr;        //READ POINTER

    reg     fifo_name_full_reg;

    reg     fifo_name_almost_full_reg;

    wire    [64-1:0]fifo_name_almost_full_val;

    reg     fifo_name_empty_reg;

    reg     fifo_name_almost_empty_reg;

    wire    [64-1:0]fifo_name_almost_empty_val;

   …    

    

    assign fifo_name_almost_full_val = 3;

    assign fifo_name_almost_empty_val = Reg1_F_q;

    …

    

    always @(posedge clk)

    …

                

            if (fifo_name_in_enb && !fifo_name_full_reg)

                begin

                    fifo_fifo_name[fifo_name_wr_ptr] <= fifo_name_in; // fifo hw write

                    fifo_name_wr_ptr  <= fifo_name_wr_ptr + 1'h1;

                if (fifo_name_wr_ptr == fifo_name_depth -1)

                    fifo_name_wr_ptr  <= 3'h0;

                end

     …    

    always @(posedge clk)

        …                

            if (fifo_name_rd_valid && !fifo_name_empty_reg)

                begin

                    fifo_name_rd_data <=  fifo_fifo_name[fifo_name_rd_ptr]; // fifo sw read

                    fifo_name_rd_ptr  <= fifo_name_rd_ptr + 1'h1;

                if (fifo_name_rd_ptr == fifo_name_depth -1)

                    fifo_name_rd_ptr  <= 3'h0;

                end

   //------------- FIFO_NAME USEDW ----------

    always @(posedge clk)

   …

   if ((fifo_name_in_enb && !fifo_name_full_reg) && (fifo_name_rd_valid && !fifo_name_empty_reg))

            fifo_name_usedW_sig <= fifo_name_usedW_sig;

   else if (fifo_name_in_enb && !fifo_name_full_reg)

            fifo_name_usedW_sig <= fifo_name_usedW_sig +1;

   else if (fifo_name_rd_valid && !fifo_name_empty_reg)

            fifo_name_usedW_sig <= fifo_name_usedW_sig -1;

    else

            fifo_name_usedW_sig <= fifo_name_usedW_sig;

    

    …

    //------------ FIFO_NAME FLAGS ------------

    always @(fifo_name_usedW_sig)

    …            

       if (fifo_name_usedW_sig == fifo_name_depth)

            fifo_name_full_reg  <= 1;

        if (fifo_name_usedW_sig == 0)

            fifo_name_empty_reg  <= 1;

        if (fifo_name_usedW_sig >= fifo_name_almost_full_val)

            fifo_name_almost_full_reg  <= 1;

        if (fifo_name_usedW_sig <= fifo_name_almost_empty_val)

            fifo_name_almost_empty_reg  <= 1;

    end   //always

        

    // FIFO_NAME OUTPUT ASSIGNMENT

    assign fifo_name_usedW = fifo_name_usedW_sig;

    assign S1 = fifo_name_almost_full_reg;

    assign fifo_name_full = fifo_name_full_reg;

   …    

endmodule
