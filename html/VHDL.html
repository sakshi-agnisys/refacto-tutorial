VHDL

IDS generates two styles of VHDL RTL, one is VHDL standard compliant and other one is VHDLalt2. Both styles of RTL generation is possible using IDS-Word/Excel and IDS-Batch.


VHDLalt1 style

This style contain records in packages. To generate the VHDL RTL in alt1 style output from IDS-Word and IDS-Excel the check box will be selected by default in the the IDS-Word/Excel, in Configuration/Outputs section.



IDSWord/IDSExcel



       

IDS-Batch

To generate VHDLalt1 RTL through IDS-Batch, output tag is “vhdl” to generate output.


idsbatch <input_file> -out “vhdl” –bus <bus_name>


VHDLalt2 style

IDS-Word/IDS-Excel


To generate the VHDL RTL in alt2 from IDS-Word and IDS-Excel a new check box is introduced in Configuration. In Configuration/Outputs section, select VHDL alt2 to generate the output.




VHDL alt1

LIBRARY IEEE;

USE IEEE.STD_LOGIC_1164.ALL;

use IEEE.std_logic_unsigned.all;

use IEEE.numeric_std.all;

use IEEE.std_logic_textio.all;

use work.Block_pkg.all;


-------------------------------------------------------------

-- BLOCK BLOCK ENTITY : Block_e

-------------------------------------------------------------


entity Block_e is

    generic (

        

        

        G_AMBA_ADDR_WIDTH : positive :=  C_AMBA_ADDR_WIDTH;

        G_AMBA_BUS_WIDTH  : positive :=  C_AMBA_BUS_WIDTH;

        

        G_Block_offset : natural := C_Block_offset

    );

    

    port (

        

        --  BUS  : AMBA

        HCLK        :    in       std_logic;                --  Bus clock

        reset_l     :    in       std_logic;                --  Bus reset      

        …

        …

        ….

        reg_in_enb       : in   Block_enb_inrec;    -- input enables for registers

        reg_out_enb      : out  Block_enb_outrec;  -- output enables for registers

        reg_in           : in   Block_inrec;      -- input register fields (all writeable register fields)

        reg_out          : out  Block_outrec;     -- output registers (all readable registers)

        

        ext_in           : in   Block_ext_inrec;          -- incoming signals for Externally implemented registers and sections

        ext_out          : out  Block_ext_outrec         -- outgoing signals for Externally implemented registers and sections

        

        

        

    );

end Block_e;

-- ARCHITECTURE : Block_e ENTITY

-------------------------------------------------------------


architecture rtl of Block_e is

    

    -- AMBA signals

    

    signal clk              :   std_logic;

    ….

    ….

    ….

    signal rd_data_vld0     :   std_logic;

    signal rd_data          :   std_logic_vector(G_AMBA_BUS_WIDTH-1   downto 0);

    

    

    signal rd_data0      :  std_logic_vector(G_AMBA_BUS_WIDTH-1 downto 0);

    

    

    -------------------------------------------

    -- signal declaration

    -------------------------------------------

    -- : REG1 Register

    

    signal decode_Reg1    : std_logic;                         -- decode signal

    

    

    --  Register : REG1

    signal rd_data_Reg1       : std_logic_vector(31 downto 0);    --  Read data signal

    …

    …    

    begin

        

        AMBA_WIDGET: entity work.amba_widget

        

        generic map (

            bus_width   => G_AMBA_BUS_WIDTH,

            addr_width  => G_AMBA_ADDR_WIDTH

        )

        port map (

            HCLK       =>   HCLK,

            …

            …

            -- XRSL IO

            clk            =>      clk,

            …

            …

        );

        

        ext_out_process : process(clk)

        begin

            if rising_edge(clk) then

                ext_out <= ext_out_d;

            end if;

        end process ext_out_process;

        

        

        

        -------------------------------------

        -- Register/Section decode

        -------------------------------------

        decode_Reg1 <= '1' when to_integer(unsigned(address(G_AMBA_ADDR_WIDTH-1 downto 0))) = (offset_Reg1 + G_Block_offset)  else '0';

        

        

        

        

        ----------------------------------------------------------------------------------------------------

        -- REGISTER      :  REG1

        -- ADDRESS       :  0x0                WIDTH : 32

        -- HW ACCESS     :  READ-WRITE

        -- SW ACCESS     :  READ-WRITE

        --

        -- FIELDS   :

        --     31:0 : Fld  ( SW : Read-Write HW : Read-Write )

        -----------------------------------------------------------------------------

        -- DESCRIPTION   :  NA

        --

        

        wr_valid_Reg1 <= '1' when  decode_Reg1 = '1' and wr_stb = '1'  else '0';

        ----------------------------------------------------------------------------

        -- FIELD : FLD

        -- HW ACCESS  :  READ-WRITE                          WIDTH  :  32

        -- SW ACCESS  :  READ-WRITE                          OFFSET :  0

        -----------------------------------------------------------------

        -- DESCRIPTION   :  NA

        --

        Reg1_Fld : process (clk)

        begin

            if rising_edge(clk) then

                if reset_l = '0' then

                    q_Reg1_Fld  <=  default_Reg1_Fld;

                else

                    if (reg_in_enb.Reg1_Fld= '1')  then         --  HW write

                        

                        q_Reg1_Fld <= reg_in.Reg1_Fld;

                    else

                        

                        if ( wr_valid_Reg1 = '1') then          --  SW Write

                            q_Reg1_Fld <= wr_data(31 downto 0);

                            

                            

                        end if;

                    end if;

                    

                end if; -- reset

                

            end if; -- clock edge

            

        end process Reg1_Fld;

        -- End REG1_FLD process

        ----------------------------------------------------------------------------

        

        

        

        ------------------------------------------

        -- signal/output assignment

        ------------------------------------------

        

        rd_data_Reg1 <=   q_Reg1_Fld;        -- Fields value concatenation of REG1 register

        

        reg_out.Reg1_Fld  <= q_Reg1_Fld;  -- HW output : FLD Read data

        

        reg_out_enb.Reg1  <= wr_valid_Reg1;-- HW output : REG1 enable

        

        

        

        -- END : REGISTER - REG1

        

        

        

        -------------------------------

        --  Read Data back stages

        --------------------------

        read_back_mux : process(

            decode_Reg1,   rd_data_Reg1

        )

        begin

            

            rd_data0     <= (others => '0');

            

            if (decode_Reg1 = '1') then

                rd_data0 <= rd_data_Reg1;

            end if;

            

        end process read_back_mux;

        

        -- End of Read Data back stages

        

        

        request <= '1';          -- HIGH(always for internal registers) indicates bus is free , LOW indicates bus is busy, external_wr_req(always for external registers)

        rd_data_vld0 <= rd_stb;  -- if all internal registers, route the read strobe back, data is already valid

        rd_wait  <= '1';         -- HIGH in case of internal registers implementation and external_rd_ack when to read external registers

        

        rd_data      <= rd_data0;

        rd_data_vld  <= rd_data_vld0;

        

        ext_out_d. placeholder <= '0';

        

        

    end rtl;

-- end_arch
