custom_error


IDS does support "custom_error=true" property which generates the error signal when proprietary bus is being used. User can use this property at the top element.


Example: IDS-NG         IDS-Word        IDS-Excel        SystemRDL


IDS-NG Register View:



IDS-NG Spreadsheet View:



SystemRDL:


property custom_error {type=boolean; component=addrmap;};

 

addrmap block1_custom {

  custom_error = true;

    reg reg1 { 

     regwidth = 32;

      field {

        hw = rw;

        sw = rw;

        onread=r;

        onwrite=w;

      }f1[31:0] = 32'h0;

    };

    reg reg2 { 

     regwidth = 32;

      field {

        hw = rw;

        sw = rw;

        onread=r;

        onwrite=w;

      }f1[31:0] = 32'h0;

    }; 

  reg1 reg1 @0x0;

  reg2 reg2 @0xA;

};
