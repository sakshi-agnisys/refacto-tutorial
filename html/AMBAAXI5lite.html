AMBA AXI5lite

To enhance the capability of the IDSBatch tool, we are supporting the new bus AMBA-AXI5LITE. AXI5Lite is a simplified version of the AMBA (Advanced Microcontroller Bus Architecture) AXI (Advanced eXtensible Interface) protocol. It is designed for simple and low-complexity systems, providing a streamlined interface for efficient communication between IP (Intellectual Property) blocks in a SoC (System-on-Chip) design.AXI5Lite specifies a subset of the full AXI protocol, with fewer features and reduced complexity. It simplifies the interconnect requirements and is commonly used for lightweight peripherals or control interfaces within a system.

module ids_chip_n_axi_aggregation(

    aclk,   // Bus clock

    aresetn,   // Reset

    awaddr,   // Write address

    awvalid,   // Write address valid : This signal indicates that write address is valid

    awready,   // Write address ready : This signal indicates that the slave is ready to accept an address

    awprot,   // Write Protection Type

    awid,   // Write address ID

    wdata,   // Write data

    wvalid,   // Write valid         : This signal indicates that valid write data and strobes are available

    wready,   // Write ready         : This signal indicates that the slave can accept the write data

    wstrb,   // Write Strobes

    bresp,   // Write Response

    bready,   // Response Ready

    bvalid,   // Response valid

    Block1_ids_wstrb,   // Write Strobes

    Block1_ids_rid,   // Read ID

    Block1_ids_rresp

    );

    

    // PARAMETERS

    parameter bus_width = 32;

    parameter addr_width = 2;

    parameter Block1_addr_width =2;

    

    input aclk;

    input aresetn;

    input [addr_width-1 : 0] awaddr;

    input awvalid;

    output awready;

    input [2 : 0] awprot;

    input [4 : 0] awid;

    input [bus_width-1 : 0] wdata;

    input wvalid;

    output wready;

    input [bus_width/8 - 1 : 0] wstrb;

    output [1 : 0] bresp;

    input bready;

    output bvalid;

    input [4 : 0] bid;

    input [addr_width-1 : 0] araddr;

    input arvalid;

    output arready;

    input [2 : 0] arprot;

    input [4 : 0] arid;

    output [bus_width-1 : 0] rdata;

    output rvalid;

    input rready;

    output [1 : 0] rresp;

    input [4 : 0] rid;

    

    // Block1 SIGNALS

    output Block1_ids_aclk;

    output Block1_ids_aresetn;

    output [Block1_addr_width-1 : 0] Block1_ids_awaddr;

    output Block1_ids_awvalid;

    input Block1_ids_awready;

    output [2 : 0]Block1_ids_awprot;

    output [bus_width-1 : 0] Block1_ids_wdata;

    output Block1_ids_wvalid;

    input Block1_ids_wready;

    output [bus_width/8-1 : 0] Block1_ids_wstrb;

    input [1 : 0]Block1_ids_bresp;

    output Block1_ids_bready;

    input Block1_ids_bvalid;

    output [Block1_addr_width-1 : 0] Block1_ids_araddr;

    output Block1_ids_arvalid;

    input Block1_ids_arready;

    output [2 : 0]Block1_ids_arprot;

    input [bus_width-1 : 0] Block1_ids_rdata;

    input Block1_ids_rvalid;

    output Block1_ids_rready;

    input [1 : 0]Block1_ids_rresp;

    output [4 : 0]Block1_ids_awid;

    input [4 : 0]Block1_ids_bid;

    output [4 : 0]Block1_ids_arid;

    input [4 : 0]Block1_ids_rid;

    .          .        .        

    .          .        .        

       

    // Block1 ASSIGN STATEMENTS

    assign Block1_ids_aclk = aclk;

    assign Block1_ids_aresetn = aresetn;

    assign Block1_ids_awaddr = awaddr[Block1_addr_width-1 : 0];

    assign Block1_ids_awvalid = awvalid;

    assign Block1_ids_awprot = awprot;

    assign Block1_ids_wdata = wdata;

    assign Block1_ids_wvalid = wvalid;

    assign Block1_ids_wstrb = wstrb;

    assign Block1_ids_bready = bready;

    assign Block1_ids_araddr = araddr[Block1_addr_width-1 : 0];

    assign Block1_ids_arvalid = arvalid;

    assign Block1_ids_arprot = arprot;

    assign Block1_ids_rready = rready;

    assign Block1_ids_awid = awid;

    assign Block1_ids_arid = arid;

endmodule
