OCP

IDS supports OCP bus both in Verilog and VHDL. All the transactions to the registers pass through this OCP widget.

OCP bus in verilog:

                           

       ocpip_widget #(.addr_width(addr_width), .bus_width(bus_width)) ocp(

            .clk(clk),

            .mreset_n(mreset_n),

            .maddr(maddr),

            .mcmd(mcmd),

            .mdata(mdata),

            .mdatabyteen(mdatabyteen),

            .scmdaccept(scmdaccept),

            .sdata(sdata),

            .sresp(sresp),

            .clock(clock),

            .reset_out(reset_l),

            .wr_data(wr_data),

            .addr(addr),

            .rd_data(rd_data),

            .request(request),

            .wr_stb(wr_stb),

            .rd_stb(rd_stb),

            .rd_data_vld(rd_data_vld),

            .byte_enb(byte_enb));

OCP Bus in VHDL :


     OCPIP_WIDGET: entity work.ocpip_widget

 

     generic map (

      bus_width   =>  G_OCP_BUS_WIDTH,

      addr_width  =>  G_OCP_ADDR_WIDTH

    )

    port map (

      --  BUS OCP

      clk             =>    clk,

      mreset_n        =>    mreset_n,

      mcmd            =>    mcmd,

      maddr           =>    maddr,

      mdata           =>    mdata,

      mdatabyteen     =>    mdatabyteen,

      scmdaccept      =>    scmdaccept,

      sresp           =>    sresp,

      sdata           =>    sdata,

      

      -- XRSL IO

      wr_data         =>   wr_data,

      rd_data         =>   rd_data,

      request         =>   request,

      rd_data_vld     =>   rd_data_vld,

      reset_out       =>   reset_l,

      byte_enb        =>   byte_enb,

      addr            =>   addr,

      wr_stb          =>   wr_stb,

      rd_stb          =>   rd_stb

    );
