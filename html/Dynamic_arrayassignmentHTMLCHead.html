Dynamic_array assignment HTML CHeader

Dynamically overriding the property at run time for a certain iteration in the register if count is applied on the register or regfile.

Benefits:

  • Users can change the default value of the register fields at run time.

How to apply dynamic array property on iteration

 

In the above structure, the highlighted portion is representing the identifiers and rest of code are keywords.

Case 1: Reset or default value of fields dynamically changed from registers.            

   addrmap block {

               reg {

                     field {} fld1[7:0] = 8'h2;

                     field {} fld2[15:8] = 8'h2;

                     field {} fld3[23:16];

                     field {} fld4[31:24];

               } csr1[4];

               csr1[1].fld2->reset = 8'ha;

               csr1[2].fld2->reset = 8'hb;

               csr1[3].fld3->reset = 8'hc;

               csr1[3].fld4->reset = 8'hd;

         };

Case 2: Reset or default value of fields dynamically changed from regfile.

             addrmap block {

                             regfile test {

                                             reg {

                                                             field {} fld1[7:0] = 8'h2;

                                                             field {} fld2[15:8] = 8'h2;

                                                             field {} fld3[23:16];

                                                             field {} fld4[31:24];

                                             } csr1[4];

                             };

                             test test[4];

                             test[0].csr1[0].fld1->reset = 8'ha;

                             test[0].csr1[2].fld1->reset = 8'hb;

                             test[1].csr1[2].fld2->reset = 8'hc;

                             test[2].csr1[1].fld4->reset = 8'hd;

                             test[2].csr1[3].fld4->reset = 8'he;

             };

#define BLOCK_TEST_CSR1_FLD1_RESET(test_i, test_csr1_i) \

#define BLOCK_TEST_CSR1_FLD2_RESET(test_i, test_csr1_i) \

         : 0x2   )

#define BLOCK_TEST_CSR1_FLD4_RESET(test_i, test_csr1_i) \
