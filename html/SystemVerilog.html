SystemVerilog

SV intefaces

(Compatible with IDS version 5.18.0 and above)


Interfaces are used for the encapsulation of ports. In IDesignSpec™, interfaces are used to bundle the hardware ports of the registers.


Advantages of IDesignSpec™ generated interfaces:-

  • Can support multidimensional array in port list.
  • Interface of each register is different.
  • Interfaces of external registers are different.


Whenever the sv output is selected, Interface of each register in a block is created in a separate file named as “<name of that block>_if.sv”. The number of interface files in the output directory is same as the no of blocks in the specification. The main System Verilog (filename.sv) and verilog (filename.v) files are also created.


The main System Verilog file contains a wrapper module, which is used to connect all interfaces with IDS generated Verilog module.

Generation of System Verilog Output from IDS


IDSBatch

To generate sv output using IDS-Batch, add "sv" output tag to -out option


IDSbatch –out sv –bus <bus>  <inputFile>

IDSWord/Excel


To generate sv output using IDS-Word/IDSexcel, select "System Verilog" option in Output Settings in configure window.

System Verilog Code


This is the code for interface file:-


 

// INTERFACE: REG_1

interface block1_reg1_interface # (

…

parameter  block1_reg11_count = 2,

parameter  block1_reg12_count = 3

);

logic  reg1_enb[block1_reg11_count-1 : 0][block1_reg12_count-1 : 0];

logic   reg1_F_in_enb[block1_reg11_count-1 : 0][block1_reg12_count-1 : 0] ;

logic  [31 : 0] reg1_F_r[block1_reg11_count-1 : 0][block1_reg12_count-1 : 0] ;

logic   [31 : 0] reg1_F_in[block1_reg11_count-1 : 0][block1_reg12_count-1 : 0] ;


modport mp (input reg1_F_in_enb, reg1_F_in,output reg1_enb, reg1_F_r);

endinterface


This the code of System Verilog file:-


`include "block1_if.sv"

`include "undefined.v"


`include "IDS_amba_if.sv"



module block1_wrapper

    #(

    //  PARAMETERS

    …

    parameter  block1_reg11_count = 2,

    

    parameter  block1_reg12_count = 3,

    

    parameter block1_address_width = addr_width

    )

    (

    // Interfaces

    block1_reg1_interface.mp   block1_reg1_if,

    IDS_amba_if.amba_mp  block1_amba_if

);

    // signals for reg1

    wire [block1_reg11_count*block1_reg12_count-1 : 0] reg1_enb;

    wire [block1_reg11_count*block1_reg12_count-1 : 0] reg1_F_in_enb ;

    wire  [block1_reg11_count*block1_reg12_count*32-1 : 0] reg1_F_r ;

    wire  [block1_reg11_count*block1_reg12_count*32-1 : 0] reg1_F_in ;

    

    // Signal Assignment

    

    generate

        genvar  block1__reg11_i ;

        for( block1__reg11_i = 0; block1__reg11_i < block1_reg11_count;block1__reg11_i = block1__reg11_i+ 1)

        begin : block1__reg11

            genvar  block1__reg12_i ;

            for( block1__reg12_i = 0; block1__reg12_i < block1_reg12_count;block1__reg12_i = block1__reg12_i+ 1)

            begin : block1__reg12

                assign  block1_reg1_if.reg1_enb[block1__reg11_i][block1__reg12_i] = reg1_enb[block1__reg11_i * block1_reg12_count + block1__reg12_i];

                assign   reg1_F_in_enb [block1__reg11_i * block1_reg12_count + block1__reg12_i] = block1_reg1_if.reg1_F_in_enb[block1__reg11_i][block1__reg12_i] ;

                

                assign   block1_reg1_if.reg1_F_r[block1__reg11_i][block1__reg12_i] =  reg1_F_r [(block1__reg11_i * block1_reg12_count + block1__reg12_i) *32+31 :  (block1__reg11_i * block1_reg12_count + block1__reg12_i) *32 ];

                

                assign    reg1_F_in [(block1__reg11_i * block1_reg12_count + block1__reg12_i) *32+31 :  (block1__reg11_i * block1_reg12_count + block1__reg12_i) *32 ] = block1_reg1_if.reg1_F_in[block1__reg11_i][block1__reg12_i];

            end

        end

    endgenerate

    

    // Module Instantiation

    

    block1_IDS   block1IDS (

    

    //block1_IDS

    

    .reg1_enb(reg1_enb),

    .reg1_F_in_enb(reg1_F_in_enb),

    .reg1_F_in(reg1_F_in),

    .reg1_F_r (reg1_F_r),

    

    

    //AMBA signals

    

    .hclk(block1_amba_if.hclk),

    .hresetn(block1_amba_if.hresetn),

    .hwrite(block1_amba_if.hwrite),

    .haddr(block1_amba_if.haddr),

    .hwdata(block1_amba_if.hwdata),

    .hrdata(block1_amba_if.hrdata),

    .hready(block1_amba_if.hready),

    .htrans(block1_amba_if.htrans),

    .hsize(block1_amba_if.hsize),

    .hresp(block1_amba_if.hresp)

    );

    

    // Instantiation End

    

endmodule
