JSON

IDesignSpecTM enables IP/SOC design teams to create code files from the register specification. IDesignSpecTM (IDS) now supports JSON as an input and this document describes how the user can generate the output from JSON input.


Example:


JSON format input:


{"block": {

    "address": 0,

    "config": {

        "addressunit": 8,

        "busdomains": {"busdomain": {

            "address": 0,

            "addressUnit": 8,

            "bus": "AMBA",

            "endaddress": 3,

            "name": "default_map",

            "offset": 0,

            "size": 4

        }},

        "buswidth": 32,

        "regwidth": 32,

        "variants": {"variant": {

            "doc": "'none' variant states including all templates which are not assigned any variant property.",

            "isselected": true,

            "name": "none"

        }}

    },

    "endaddress": 3,

    "name": "Block1",

    "offset": 0,

    "reg": {

        "address": 0,

        "config": {

            "busdomains": {"busdomain": {

                "address": 0,

                "addressUnit": 8,

                "bus": "AMBA",

                "endaddress": 3,

                "name": "default_map",

                "offset": 0,

                "size": 4

            }},

            "regwidth": 32

        },

        "endaddress": 3,

        "field": {

            "default": 0,

            "hw": "rw",

            "name": "F1",

            "offset": "31:0",

            "sw": "rw"

        },

        "name": "Reg1",

        "offset": 0,

        "size": 4

    },

    "size": 4

}}


Command line:


IDSbatch test.json -out verilog -bus apb -dir IDS 


Generated Output:


module Block1_IDS(

    

    

    // REGISTER : REG1 PORT SIGNAL

    Reg1_enb,

    Reg1_F1_in,

    Reg1_F1_in_enb,

    Reg1_F1_r,

    

    //APB signals

    pclk,   // Bus clock

    presetn,   // Reset

    psel,   // Select    : It indicates that the target device is selected and a data transfer is required

    penable,   // Enable    : This signal indicates the second and subsequent cycles of an APB transfer

    pwrite,   // Direction : This signal indicates an APB write access when HIGH and an APB read access when LOW

    pprot,   // Protection type : This signal indicates the normal, privileged, or secure protection level of the transaction

    pstrb,   // Write strobes : This signal indicates which byte lanes to update during a write transfer

    pwdata,   // Write data

    paddr,   // Address bus

    pready,   // Ready     : The target uses this signal to extend an APB transfer

    prdata,   // Read data

    pslverr     //pslverr : This signal indicates a transfer failure.

    );

    

    

    // PARAMETERS

    

    parameter bus_width = 32;

    parameter addr_width = 2;

    

    parameter block_size  = 'h4;

    

    parameter [addr_width-1 : 0] block_offset = {(addr_width){1'b0}};

    

    

    

    // REGISTER  : REG1 SIGNALS

    wire Reg1_decode;        // Write Decode

    wire Reg1_wr_valid;        // Write Valid

    wire [63 : 0] Reg1_offset;        // Offset

    wire Reg1_rd_valid;        // Read Valid

    wire [bus_width-1 : 0] Reg1_rd_data;        // Read Data

    output Reg1_enb; // REGISTER ENABLE

    reg [31 : 0] Reg1_F1_q; // FIELD : F1

    input [32- 1 : 0] Reg1_F1_in;

    input Reg1_F1_in_enb;

    output [31 : 0] Reg1_F1_r;

    

    

    //APB signals

    input pclk;

    input presetn;

    input psel;

    input penable;

    input pwrite;

    input [2 : 0] pprot;

    input [bus_width/8-1 : 0] pstrb;

    input [bus_width-1 : 0] pwdata;

    input [addr_width-1 : 0] paddr;

    output pready;

    output [bus_width-1 : 0] prdata;

    output pslverr;

    wire [2 : 0] pprot_i;

    

    

    wire [bus_width-1 : 0] reg_enb;

    wire clk;

    wire reset_l;

    wire rd_stb;

    wire rd_wait;

    wire wr_stb;

    wire [addr_width-1 : 0] address;

    wire  [bus_width/8 -1 : 0] byte_enb;

    wire [bus_width-1 : 0] wr_data;

    wire request;

    wire [bus_width-1 : 0] rd_data;

    wire rd_data_vld;

    wire error;

    

    apb_widget # (.addr_width(addr_width), .bus_width(bus_width) )apb (

            .pclk(pclk),

            .presetn(presetn),

            .pwdata(pwdata),

            .paddr(paddr),

            .prdata(prdata),

            .pready(pready),

            .pwrite(pwrite),

            .pprot(pprot),

            .pprot_i(pprot_i),

            .pstrb(pstrb),

            .psel(psel),

            .penable(penable),

            .pslverr(pslverr),

            

            .clk(clk),

            .reset_l(reset_l),

            .request(request),

            .wr_stb(wr_stb),

            .rd_stb(rd_stb),

            .rd_data(rd_data),

            .wr_data(wr_data),

            .rd_wait(rd_wait),

            .address(address),

            .rd_data_vld(rd_data_vld),

            .byte_enb(byte_enb),

            .error(error));

    // end widget

    

    assign reg_enb = {

    {8{byte_enb[3]}} ,

    {8{byte_enb[2]}} ,

    {8{byte_enb[1]}} ,

    {8{byte_enb[0]}}};

    

    

    //----------------------------------------------------------------------------------------------------

    // REGISTER      :  REG1

    // ADDRESS       :  block_offset +'h0 WIDTH : 32

    // HW ACCESS     :  RW

    // SW ACCESS     :  READ-WRITE

    //

    // FIELDS   :

    //     31:0 : F1  ( SW : RW HW : RW )

    //-----------------------------------------------------------------------------

    /* DESCRIPTION  :  NA

    */

    assign Reg1_wr_valid = Reg1_decode && wr_stb  ;

    assign Reg1_offset = block_offset +'h0;

    assign Reg1_decode = (address[addr_width-1 : 0] == Reg1_offset[addr_width-1 : 0]) ? 1'b1 : 1'b0;

    assign Reg1_rd_valid = Reg1_decode && rd_stb ;

    assign Reg1_enb = Reg1_wr_valid;

    

    //----------------------------------------------------------------------------

    // FIELD  : F1

    // HW ACCESS  : RW                          WIDTH : 32

    // SW ACCESS  : RW                          OFFSET : 0

    //-----------------------------------------------------------------

    /*DESCRIPTION    : NA

    */

    

    always @(posedge clk)  begin

    if (!reset_l)

        begin

            Reg1_F1_q <= 32'd0;

        end

    else

        begin

            

        if (Reg1_F1_in_enb) //F1 : HW Write

            begin

                Reg1_F1_q <= Reg1_F1_in;

            end

        else

            begin

            if (Reg1_wr_valid) //F1 : SW Write

                begin

                    Reg1_F1_q <= (wr_data [31 : 0]  & reg_enb  [31 : 0] ) | (Reg1_F1_q & (~reg_enb  [31 : 0] ));

                end

            end

        end

    end //end always

    

    assign Reg1_F1_r = Reg1_F1_q; // Field : F1

    assign Reg1_rd_data  = Reg1_rd_valid ? {Reg1_F1_q} : 32'd0;

    

    

    

    assign rd_data_vld = rd_stb;

    assign rd_data = Reg1_rd_data ;

    assign request = 1'b1;

    assign rd_wait = 1'b1;

    

    assign error = 1'b0;

    

endmodule


Support for hex address and json_show_inst in JSON output


IDS-Batch currently offers support for JSON output format, presenting values like "address," "address2," "endaddress," and "offset" in decimal format, This improvement involves introducing the "json_num_hex=true" at the top level. This addition ensures that these values, excluding field data, are displayed in hexadecimal format across all components beneath the top level. 

Currently in JSON output format, in case of “repeat” property on any component, repeated instances are not displayed individually. This improvement involves introducing the property “json_show_inst=true” at the top level. When applied, repeat instances are displayed individually in the form of an array.

To support this, the top property is created “json_num_hex= true" to be used in the batch command. This option will display the values of  "address", "address2", "endaddress", and "offset" in hexadecimal format and “json_show_inst=true” added in the batch command. repeat instances are displayed individually in the form of an array [].


SystemRDL


property chip{type = boolean; component = addrmap;};


addrmap chip1{

chip = true;

   addrmap block1{

     regfile rf{

      reg r1{

       field f1{

       sw= rw;

       hw=rw;

       }f1[31:0];

     };r1 r1[2];

   }; rf rf[2];

 }; block1 block1[2];

 };


Command Line:


 idsbatch <input_file> -out json -bus amba

Support for default properties


A new feature has been introduced in which by providing a JSON file that will contain all the properties employed in the input test case along with their default value as well as the actual value that is being used in the input test case.


IDS-NG Input:



SystemRDL:


property chip {type = boolean; component = addrmap;};

property clock_edge {type = string; component = addrmap|reg;};

property cheader_name_format {type = string; component = addrmap;};

Change chip_name to top_element in profile.json

We've made a change in  profile.json file that contains comprehensive information about output memory and time for all processes. Additionally, when we  use the -profile switch, the profile.json file will be generated automatically. In the file  “profile.json” the key "chip_name" is changed to "top_element".

SystemRDL Input:

property chip {type = string; component = addrmap;};

addrmap chip_1 {

  chip = "true";

  addrmap block_1 {

    reg reg1 {

      field f1 {

        sw=rw;

        hw=rw;

      } f1[31:0]=0;

    }; reg1 reg1;

  };

  block_1 block_1;

  addrmap block_2 {

    reg reg2 {

      field f2 {

        sw=rw;

        hw=rw;

      } f2[31:0]=0;

    };

    reg2 reg2;

  };

  block_2 block_2;

};


addrmap chip_top {

  chip = "true";

  chip_1 chip1;

  chip_1 chip2;

};

idsbatch input.rdl -o json -bus apb -dir profile.json -profile
